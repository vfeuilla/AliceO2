143  void process(aod::Collision const& theCollision,
144  aod::V0Datas const& theV0s,
145  tracksAndTPCInfo const& theTracks,
146  aod::McParticles const& theMcParticles)
147  {
148  for (auto& lV0 : theV0s) {
149 
150  fillHistogramsBeforeCuts(lV0);
151 
152  auto lTrackPos = lV0.template posTrack_as<tracksAndTPCInfo>(); //positive daughter
153  auto lTrackNeg = lV0.template negTrack_as<tracksAndTPCInfo>(); //negative daughter
154 
155  // apply track cuts
156  if (!(trackPassesCuts(lTrackPos) && trackPassesCuts(lTrackNeg))) {
157  continue;
158  }
159 
160  float lV0CosinePA = RecoDecay::CPA(array{theCollision.posX(), theCollision.posY(), theCollision.posZ()}, array{lV0.x(), lV0.y(), lV0.z()}, array{lV0.px(), lV0.py(), lV0.pz()});
161 
162  // apply photon cuts
163  if (!passesPhotonCuts(lV0, lV0CosinePA)) {
164  continue;
165  }
166 
167  fillHistogramsAfterCuts(lV0, lTrackPos, lTrackNeg, lV0CosinePA);
168 
169  processTruePhotons(lV0, lTrackPos, lTrackNeg, theMcParticles);
170  }
171  }

173  template <typename T>
174  void fillHistogramsBeforeCuts(const T& theV0)
175  {
176  // fill some QA histograms before any cuts
177  registry.fill(HIST("beforeCuts/hPtRec"), theV0.pt());
178  registry.fill(HIST("beforeCuts/hEtaRec"), theV0.eta());
179  registry.fill(HIST("beforeCuts/hPhiRec"), theV0.phi());
180  registry.fill(HIST("beforeCuts/hConvPointR"), theV0.v0radius());
181  registry.fill(HIST("IsPhotonSelected"), kPhotonIn);
182  }

184  template <typename T>
185  bool trackPassesCuts(const T& theTrack)
186  {
187 
188  // single track eta cut
189  if (TMath::Abs(theTrack.eta()) > fEtaCut) {
190  registry.fill(HIST("IsPhotonSelected"), kTrackEta);
191  return kFALSE;
192  }
193 
194  // single track pt cut
195  if (theTrack.pt() < fSinglePtCut) {
196  registry.fill(HIST("IsPhotonSelected"), kTrackPt);
197  return kFALSE;
198  }
199 
200  if (!(selectionPIDTPC_track(theTrack))) {
201  return kFALSE;
202  }
203 
204  if (theTrack.tpcFoundOverFindableCls() < fMinTPCFoundOverFindableCls) {
205  registry.fill(HIST("IsPhotonSelected"), kTPCFoundOverFindableCls);
206  return kFALSE;
207  }
208 
209  if (theTrack.tpcCrossedRowsOverFindableCls() < fMinTPCCrossedRowsOverFindableCls) {
210  registry.fill(HIST("IsPhotonSelected"), kTPCCrossedRowsOverFindableCls);
211  return kFALSE;
212  }
213 
214  return kTRUE;
215  }
216 
217  template <typename T>
218  bool passesPhotonCuts(const T& theV0, float theV0CosinePA)
219  {
220  if (theV0.v0radius() < fMinR || theV0.v0radius() > fMaxR) {
221  registry.fill(HIST("IsPhotonSelected"), kV0Radius);
222  return kFALSE;
223  }
224 
225  if (!ArmenterosQtCut(theV0.alpha(), theV0.qtarm(), theV0.pt())) {
226  registry.fill(HIST("IsPhotonSelected"), kArmenteros);
227  return kFALSE;
228  }
229 
230  if (TMath::Abs(theV0.psipair()) > fPsiPairCut) {
231  registry.fill(HIST("IsPhotonSelected"), kPsiPair);
232  return kFALSE;
233  }
234 
235  if (theV0CosinePA < fCosPAngleCut) {
236  registry.fill(HIST("IsPhotonSelected"), kCosinePA);
237  return kFALSE;
238  }
239 
240  return kTRUE;
241  }

243  template <typename TV0, typename TTRACK>
244  void fillHistogramsAfterCuts(const TV0& theV0, const TTRACK& theTrackPos, const TTRACK& theTrackNeg, float theV0CosinePA)
245  {
246  registry.fill(HIST("IsPhotonSelected"), kPhotonOut);
247 
248  registry.fill(HIST("hPtRec"), theV0.pt());
249  registry.fill(HIST("hEtaRec"), theV0.eta());
250  registry.fill(HIST("hPhiRec"), theV0.phi());
251  registry.fill(HIST("hConvPointR"), theV0.v0radius());
252 
253  registry.fill(HIST("hTPCdEdxSigEl"), theTrackNeg.p(), theTrackNeg.tpcNSigmaEl());
254  registry.fill(HIST("hTPCdEdxSigEl"), theTrackPos.p(), theTrackPos.tpcNSigmaEl());
255  registry.fill(HIST("hTPCdEdxSigPi"), theTrackNeg.p(), theTrackNeg.tpcNSigmaPi());
256  registry.fill(HIST("hTPCdEdxSigPi"), theTrackPos.p(), theTrackPos.tpcNSigmaPi());
257 
258  registry.fill(HIST("hTPCdEdx"), theTrackNeg.p(), theTrackNeg.tpcSignal());
259  registry.fill(HIST("hTPCdEdx"), theTrackPos.p(), theTrackPos.tpcSignal());
260 
261  registry.fill(HIST("hTPCFoundOverFindableCls"), theTrackNeg.tpcFoundOverFindableCls());
262  registry.fill(HIST("hTPCFoundOverFindableCls"), theTrackPos.tpcFoundOverFindableCls());
263 
264  registry.fill(HIST("hTPCCrossedRowsOverFindableCls"), theTrackNeg.tpcCrossedRowsOverFindableCls());
265  registry.fill(HIST("hTPCCrossedRowsOverFindableCls"), theTrackPos.tpcCrossedRowsOverFindableCls());
266 
267  registry.fill(HIST("hArmenteros"), theV0.alpha(), theV0.qtarm());
268  registry.fill(HIST("hPsiPtRec"), theV0.psipair(), theV0.pt());
269 
270  registry.fill(HIST("hCosPAngle"), theV0CosinePA);
271  }

273  template <typename TV0, typename TTRACK, typename TMC>
274  void processTruePhotons(const TV0& theV0, const TTRACK& theTrackPos, const TTRACK& theTrackNeg, const TMC& theMcParticles)
275  {
276  // todo: verify it is enough to check only mother0 being equal
277  if (theTrackPos.mcParticle().mother0() > -1 &&
278  theTrackPos.mcParticle().mother0() == theTrackNeg.mcParticle().mother0()) {
279  auto lMother = theMcParticles.iteratorAt(theTrackPos.mcParticle().mother0());
280 
281  if (lMother.pdgCode() == 22) {
282 
283  registry.fill(HIST("resolutions/hPtRes"), theV0.pt() - lMother.pt());
284  registry.fill(HIST("resolutions/hEtaRes"), theV0.eta() - lMother.eta());
285  registry.fill(HIST("resolutions/hPhiRes"), theV0.phi() - lMother.phi());
286 
287  TVector3 lConvPointRec(theV0.x(), theV0.y(), theV0.z());
288  TVector3 lPosTrackVtxMC(theTrackPos.mcParticle().vx(), theTrackPos.mcParticle().vy(), theTrackPos.mcParticle().vz());
289  // take the origin of the positive mc track as conversion point (should be identical with negative, verified this on a few photons)
290  TVector3 lConvPointMC(lPosTrackVtxMC);
291 
292  registry.fill(HIST("resolutions/hConvPointRRes"), lConvPointRec.Perp() - lConvPointMC.Perp());
293  registry.fill(HIST("resolutions/hConvPointAbsoluteDistanceRes"), TVector3(lConvPointRec - lConvPointMC).Mag());
294  }
295  }

298  Bool_t ArmenterosQtCut(Double_t theAlpha, Double_t theQt, Double_t thePt)
299  {
300  // in AliPhysics this is the cut for if fDo2DQt && fDoQtGammaSelection == 2
301  Float_t lQtMaxPtDep = fQtPtMax * thePt;
302  if (lQtMaxPtDep > fQtMax) {
303  lQtMaxPtDep = fQtMax;
304  }
305  if (!(TMath::Power(theAlpha / fMaxPhotonAsymmetry, 2) + TMath::Power(theQt / lQtMaxPtDep, 2) < 1)) {
306  return kFALSE;
307  }
308  return kTRUE;
309  }

311  template <typename T>
312  bool selectionPIDTPC_track(const T& theTrack)
313  {
314  // TPC Electron Line
315  if (theTrack.tpcNSigmaEl() < fPIDnSigmaBelowElectronLine || theTrack.tpcNSigmaEl() > fPIDnSigmaAboveElectronLine) {
316  registry.fill(HIST("IsPhotonSelected"), kElectronPID);
317  return kFALSE;
318  }
319 
320  // TPC Pion Line
321  if (theTrack.p() > fPIDMinPnSigmaAbovePionLine) {
322  // low pt Pion rej
323  if (theTrack.p() < fPIDMaxPnSigmaAbovePionLine) {
324  if (theTrack.tpcNSigmaEl() > fPIDnSigmaBelowElectronLine && theTrack.tpcNSigmaEl() < fPIDnSigmaAboveElectronLine && theTrack.tpcNSigmaPi() < fPIDnSigmaAbovePionLine) {
325  registry.fill(HIST("IsPhotonSelected"), kPionRejLowMom);
326  return kFALSE;
327  }
328  }
329  // High Pt Pion rej
330  else {
331  if (theTrack.tpcNSigmaEl() > fPIDnSigmaBelowElectronLine && theTrack.tpcNSigmaEl() < fPIDnSigmaAboveElectronLine && theTrack.tpcNSigmaPi() < fPIDnSigmaAbovePionLineHighP) {
332  registry.fill(HIST("IsPhotonSelected"), kPionRejHighMom);
333  return kFALSE;
334  }
335  }
336  }
337  return kTRUE;
338  }
